ProductData Schema

The core data structure returned by all ShopGraph extraction endpoints.

Interface

ProductData
interface ProductData {
  title: string;              // Product name
  price: number | null;       // Numeric price (e.g. 98.00)
  currency: string | null;    // ISO 4217 currency code (e.g. "USD")
  brand: string | null;       // Brand or manufacturer name
  description: string | null; // Product description
  image: string | null;       // Primary image URL
  images: string[];           // All product image URLs
  availability: string | null; // Schema.org availability enum
  sku: string | null;         // SKU or product identifier
  gtin: string | null;        // GTIN/UPC/EAN barcode
  mpn: string | null;         // Manufacturer part number
  category: string | null;    // Product category/breadcrumb
  url: string;                // Canonical product URL
  rating: number | null;      // Average rating (0-5)
  review_count: number | null; // Number of reviews
  variants: Variant[];        // Product variants (size, color, etc.)

  // B2B fields (when detected)
  moq: number | null;         // Minimum order quantity
  lead_time: string | null;   // Estimated delivery/lead time
  bulk_pricing: BulkTier[];   // Volume discount tiers
}
